package Easy;

public class LargestNumber {

	
	    public static void main(String[] args) {
	        
	        int[] numbers = {10, 20, 30, 10, 20};  

	        int largest = numbers[0];  

	        for (int num : numbers) {
	            if (num > largest) {
	                largest = num;
	            }
	        }

	        System.out.println("The largest number is: " + largest);
	    }
	

	
	
	
	
	
}
